pipeline {
    agent any
    stages {
        stage('Get Release Version') {
            steps {
                script {
                    def releaseDate = sh(script: 'date +%Y%m%d', returnStdout: true).trim()
                    env.RELEASE_VERSION = "1.0.${releaseDate}"
                    echo "Release Version: ${env.RELEASE_VERSION}"
                }
            }
        }
    }
}
