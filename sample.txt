from fastapi import FastAPI
from fastapi.responses import FileResponse
from fastapi.staticfiles import StaticFiles

app = FastAPI()

# Static file directory (optional if you have multiple files to serve)
app.mount("/files", StaticFiles(directory="files"), name="files")

# Endpoint to download the script
@app.get("/download-script", summary="Download Script", description="Click to download the script.py file.")
def download_script():
    file_path = "files/script.py"
    return FileResponse(path=file_path, filename="script.py", media_type="application/octet-stream")
