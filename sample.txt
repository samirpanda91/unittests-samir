def filter_text_file(input_file, output_file):
    with open(input_file, 'r') as infile:
        lines = infile.readlines()
    
    keep_lines = []
    skip = False
    delete_block = False

    for line in lines:
        if line.strip().startswith('// test'):
            if skip and delete_block:
                skip = False
                delete_block = False
                continue  # Skip this `// test` line as it ends the block
            skip = True  # Start skipping when encountering `// test`
        
        if 'Some data 2' in line and skip:
            delete_block = True  # Mark delete block if 'Some data 2' is encountered while skipping
        
        if not skip:
            keep_lines.append(line)

    with open(output_file, 'w') as outfile:
        outfile.writelines(keep_lines)


# Usage
filter_text_file('input.txt', 'output.txt')
