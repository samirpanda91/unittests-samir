#!/bin/bash

# Replace with your script name or part of the command that starts it
PROCESS_NAME="your_python_script.py"

# Count the number of running processes
PROCESS_COUNT=$(pgrep -f "$PROCESS_NAME" | wc -l)

if [ "$PROCESS_COUNT" -gt 0 ]; then
  echo "There are $PROCESS_COUNT instances of $PROCESS_NAME running."
else
  echo "No processes found for $PROCESS_NAME."
fi
