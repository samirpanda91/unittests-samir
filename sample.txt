import pytest
from your_module import to_snake, get_hostname  # Replace 'your_module' with the actual module name

def test_to_snake_camel_case():
    assert to_snake("camelCaseWord") == "camel_case_word"

def test_to_snake_mixed_case():
    assert to_snake("MiXeD_CaSe_WoRd") == "mixed_case_word"

def test_to_snake_all_caps():
    assert to_snake("ALL_CAPS_WORD") == "all_caps_word"

def test_to_snake_already_snake_case():
    assert to_snake("snake_case_word") == "snake_case_word"

def test_to_snake_empty_string():
    assert to_snake("") == ""

def test_get_hostname_vdi():
    with patch('socket.gethostname', return_value='VDI_HOSTNAME'):
        assert get_hostname() == "127.0.0.1"

def test_get_hostname_cpc():
    with patch('socket.gethostname', return_value='CPC_HOSTNAME'):
        assert get_hostname() == "127.0.0.1"

def test_get_hostname_regular():
    with patch('socket.gethostname', return_value='regular_hostname'):
        assert get_hostname() == "regular_hostname.wellsfargo.com"
